## Step 5: Set Up Environment Variables

### 5.1 Get OAuth Credentials

#### Google OAuth:
1. Go to [Google Cloud Console](https://console.cloud.google.com/)
2. Create a new project (or use existing)
3. Enable "Google+ API"
4. Go to "Credentials" → "Create Credentials" → "OAuth 2.0 Client ID"
5. Application type: "Web application"
6. Authorized redirect URIs:
   - `http://localhost:3000/api/auth/callback/google` (development)
   - `https://yourdomain.com/api/auth/callback/google` (production)
7. Copy Client ID and Client Secret

#### Apple OAuth (for iOS):
1. Go to [Apple Developer Portal](https://developer.apple.com/)
2. Certificates, Identifiers & Profiles → Identifiers → App IDs
3. Create new Service ID
4. Enable "Sign in with Apple"
5. Configure domains and redirect URLs
6. Generate Client Secret (JWT-based)

#### GitHub OAuth (optional):
1. Go to GitHub Settings → Developer settings → OAuth Apps
2. New OAuth App
3. Homepage URL: `http://localhost:3000`
4. Callback URL: `http://localhost:3000/api/auth/callback/github`
5. Copy Client ID and Client Secret

### 5.2 Update `.env.local`

```bash
# Convex (auto-generated)
CONVEX_DEPLOYMENT=your-deployment-name
NEXT_PUBLIC_CONVEX_URL=https://your-deployment.convex.cloud

# Google OAuth
AUTH_GOOGLE_ID=your_google_client_id.apps.googleusercontent.com
AUTH_GOOGLE_SECRET=your_google_client_secret

# Apple OAuth
AUTH_APPLE_ID=your.apple.service.id
AUTH_APPLE_SECRET=your_apple_client_secret

# GitHub OAuth (optional)
AUTH_GITHUB_ID=your_github_client_id
AUTH_GITHUB_SECRET=your_github_client_secret

# Auth secret (generate with: openssl rand -base64 32)
AUTH_SECRET=your_random_32_character_secret
```

## Step 7: Set Up Convex Provider in Next.js

### 7.1 Create Convex Provider

```tsx
// app/providers/ConvexClientProvider.tsx
"use client";

import { ConvexProvider, ConvexReactClient } from "convex/react";
import { ReactNode } from "react";

const convex = new ConvexReactClient(process.env.NEXT_PUBLIC_CONVEX_URL!);

export function ConvexClientProvider({ children }: { children: ReactNode }) {
  return <ConvexProvider client={convex}>{children}</ConvexProvider>;
}
```

### 7.2 Wrap Your App

```tsx
// app/layout.tsx
import { ConvexClientProvider } from "./providers/ConvexClientProvider";

export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <html lang="en">
      <body>
        <ConvexClientProvider>
          {children}
        </ConvexClientProvider>
      </body>
    </html>
  );
}
```

## Step 8: Implement Auth UI Components

### 8.1 Sign Up Page

```tsx
// app/signup/page.tsx
"use client";

import { useState } from "react";
import { useRouter } from "next/navigation";
import { useMutation } from "convex/react";
import { api } from "@/convex/_generated/api";

export default function SignUpPage() {
  const router = useRouter();
  const createProfile = useMutation(api.profiles.createProfile);
  
  const [formData, setFormData] = useState({
    name: "",
    email: "",
    password: "",
  });

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    
    try {
      // 1. Create auth user via Convex Auth
      // (This part uses Convex Auth's signIn function)
      const { userId } = await signIn("credentials", formData);
      
      // 2. Create profile
      await createProfile({
        userId,
        name: formData.name,
        email: formData.email,
      });
      
      // 3. Redirect to onboarding
      router.push("/onboarding");
    } catch (error) {
      console.error("Signup failed:", error);
    }
  };

  return (
    <form onSubmit={handleSubmit} className="max-w-md mx-auto p-6">
      <h1 className="text-2xl font-bold mb-6">Sign Up for stars.guide</h1>
      
      <input
        type="text"
        placeholder="Full Name"
        value={formData.name}
        onChange={(e) => setFormData({ ...formData, name: e.target.value })}
        className="w-full p-2 mb-4 border rounded"
        required
      />
      
      <input
        type="email"
        placeholder="Email"
        value={formData.email}
        onChange={(e) => setFormData({ ...formData, email: e.target.value })}
        className="w-full p-2 mb-4 border rounded"
        required
      />
      
      <input
        type="password"
        placeholder="Password"
        value={formData.password}
        onChange={(e) => setFormData({ ...formData, password: e.target.value })}
        className="w-full p-2 mb-4 border rounded"
        required
      />
      
      <button
        type="submit"
        className="w-full bg-primary text-white p-2 rounded hover:bg-primary/90"
      >
        Create Account
      </button>
      
      {/* OAuth Buttons */}
      <div className="mt-6 space-y-3">
        <button
          type="button"
          onClick={() => signIn("google")}
          className="w-full border p-2 rounded flex items-center justify-center gap-2"
        >
          Continue with Google
        </button>
        
        <button
          type="button"
          onClick={() => signIn("apple")}
          className="w-full border p-2 rounded flex items-center justify-center gap-2"
        >
          Continue with Apple
        </button>
      </div>
    </form>
  );
}
```

## Troubleshooting

### Issue: "Table not found: profiles"
**Solution:** Run `npx convex dev` to sync schema

### Issue: OAuth redirect mismatch
**Solution:** Ensure redirect URIs in OAuth provider settings match exactly:
- Dev: `http://localhost:3000/api/auth/callback/google`
- Prod: `https://yourdomain.com/api/auth/callback/google`

### Issue: "AUTH_SECRET is not set"
**Solution:** Generate secret: `openssl rand -base64 32` and add to `.env.local`

### Issue: User created but profile missing
**Solution:** Ensure `createProfile` mutation is called after successful signup
